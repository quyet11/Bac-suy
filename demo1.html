$(document).ready(function () { let data = {}; // Đọc dữ liệu từ data.json và hiển thị danh sách lớp vào dropdown $.getJSON("data.json", function (response) { data = response; const themLop = $("#lop"); data.tenLop.forEach((cls, index) => { themLop.append(new Option(cls.tenLop, index)); }) }); // Xử lý sự kiện khi chọn lớp từ dropdown $("#lop").change(function () { const chonLop = $(this).val(); if (chonLop !== "") { const chonLopMoi = data.tenLop[chonLop]; hienThiBang(chonLopMoi.sinhVien); } }); function hienThiBang(sinhVien) { const tbody = $("#bangSinhVien tbody"); tbody.empty(); sinhVien.forEach((sv, index) => { const row = ` ${sv.hoTen} ${sv.maSV} ${sv.ngaySinh} ` tbody.append(row); }) } // Các hàm xử lý sự kiện thêm, sửa, xóa sinh viên });
